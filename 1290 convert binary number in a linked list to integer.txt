/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    int getDecimalValue(ListNode* head) {
        int ans=0;
        ListNode*temp=head;
        int count=0;
        count=count-1;
        while(temp!=NULL)
        {
            count++;
            temp=temp->next;
        }
        temp=head;
        while(temp!=NULL)
        {
            ans=ans+(temp->val)*pow(2,count);
            count--;
            temp=temp->next;
        }
        return ans;
    }
};
