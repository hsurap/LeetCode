class Solution {
public:
    int peakIndexInMountainArray(vector<int>& arr)
    {
        int max=-1;
        int maxi=-1;
        for(int i=0;i<arr.size();i++)
        {
            if(arr[i]>max)
            {
                max=arr[i];
                maxi=i;
                
            }
        }
        return maxi;
    }
};
